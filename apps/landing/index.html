AI-Native Design

LLMs call a fluent API instead of wrestling with raw XML. A compact intermediate format makes the entire diagram fit in a single prompt — AI generates it, the SDK validates and renders it.

const compact = compactify(defs)
const xml = Bpmn.export(expand(compact))

With BPMN SDK 
import { Bpmn } from "@bpmn-sdk/core";

const xml = Bpmn.export(
  Bpmn.createProcess("my-flow")
    .startEvent("start")
    .serviceTask("task", {
      name: "Do Something",
      taskType: "my-worker",
    })
    .endEvent("end")
    .withAutoLayout()
    .build()
);

// ✓ Valid BPMN 2.0 XML
// ✓ Auto-layout applied
// ✓ Zeebe extensions set

order-validation.ts 
import { Bpmn } from "@bpmn-sdk/core";

const defs = Bpmn.createProcess("order-validation")
  .name("Order Validation")
  .startEvent("start", { name: "Order Received" })
  .serviceTask("validate", {
    name: "Validate Order",
    taskType: "validate-order",
  })
  .serviceTask("notify", {
    name: "Send Confirmation",
    taskType: "send-email",
  })
  .endEvent("end", { name: "Done" })
  .withAutoLayout()
  .build();

const xml = Bpmn.export(defs);
approval-flow.ts 
const defs = Bpmn.createProcess("approval-flow")
  .name("Approval Workflow")
  .startEvent("start", { name: "Request Submitted" })
  .serviceTask("review", {
    name: "Auto Review",
    taskType: "auto-review",
  })
  .exclusiveGateway("decision", { name: "Approved?" })
    .branch("approved", (b) =>
      b.condition("= approved = true")
        .serviceTask("process", {
          name: "Process Request",
          taskType: "process",
        })
        .endEvent("done", { name: "Completed" })
    )
    .branch("rejected", (b) =>
      b.defaultFlow()
        .serviceTask("reject", {
          name: "Send Rejection",
          taskType: "reject",
        })
        .endEvent("rejected", { name: "Rejected" })
    )
  .withAutoLayout()
  .build();

ai-support-agent.ts 
const defs = Bpmn.createProcess("ai-support-agent")
  .name("AI Support Agent")
  .startEvent("start", { name: "Ticket Created" })
  .serviceTask("enrich", {
    name: "Fetch Context",
    taskType: "fetch-customer-data",
  })
  .adHocSubProcess("agent", (sub) => {
    sub
      .serviceTask("search", {
        name: "Search KB",
        taskType: "knowledge-search",
      })
      .serviceTask("draft", {
        name: "Draft Response",
        taskType: "draft-reply",
      });
  }, { name: "AI Agent",
       taskDefinition: { type: "io.camunda:ai-agent:1" } })
  .exclusiveGateway("check", { name: "Confidence?" })
    .branch("high", (b) =>
      b.condition("= confidence > 0.9")
        .serviceTask("send", { name: "Auto-Reply",
          taskType: "send-reply" })
        .endEvent("resolved", { name: "Resolved" })
    )
    .branch("low", (b) =>
      b.defaultFlow()
        .userTask("escalate", { name: "Human Review",
          formId: "review-form" })
        .endEvent("escalated", { name: "Escalated" })
    )
  .withAutoLayout()
  .build();
Quickstart

Up and running
in 3 steps

01

Install

npm install @bpmn-sdk/core
yarn add @bpmn-sdk/core
pnpm add @bpmn-sdk/core
bun add @bpmn-sdk/core
02

Create a process

import { Bpmn } from "@bpmn-sdk/core";

const xml = Bpmn.export(
  Bpmn.createProcess("hello")
    .startEvent("start")
    .serviceTask("task", {
      name: "Hello World",
      taskType: "greet",
    })
    .endEvent("end")
    .withAutoLayout()
    .build()
);
03

Deploy & run

import { Engine } from "@bpmn-sdk/engine";

const engine = new Engine();
await engine.deploy({ bpmn: xml });

engine.registerJobWorker(
  "greet",
  async (job) => {
    console.log("Hello!");
    await job.complete();
  }
);
engine.start("hello");
